üíª How to Use

With R (SQLite)

library(DBI)
library(RSQLite)

# Connect to database
con <- dbConnect(RSQLite::SQLite(), "database.db")

# View available tables
dbListTables(con)

# Query documents
documents <- dbGetQuery(con, "SELECT * FROM documents LIMIT 10")

# Search for specific terms
climate_docs <- dbGetQuery(con,
"SELECT celex_id, title, word_count FROM documents
WHERE full_text LIKE '%climate%' AND scrape_status = 'success'")

With Python (API)

import requests
import json

# Fetch data from API
response = requests.get('https://eurlex-prelim-database-production.up.railway.app/api/data')
documents = response.json()

# Process documents
for doc in documents:
    print(f"Document: {doc['celex_id']}")
    print(f"Title: {doc['title']}")
    print(f"Words: {doc['word_count']}")

With Python (SQLite)

import sqlite3
import pandas as pd

# Connect to downloaded database
conn = sqlite3.connect('database.db')

# Load data into pandas
df = pd.read_sql_query("SELECT * FROM documents", conn)

# Basic exploration
print(df.shape)
print(df.columns)
print(df['document_type'].value_counts())

üìã Dataset Schema

Documents Table

  • celex_id - Unique EU document identifier
  • title - Document title
  • document_type - Type (Regulation, Directive, etc.)
  • date_published - Publication date
  • author - Authoring institution
  • full_text - Complete document content
  • word_count - Number of words in document
  • url - Original EUR-Lex URL
  • scrape_status - Data collection status
